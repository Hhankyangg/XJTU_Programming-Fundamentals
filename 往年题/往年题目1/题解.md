````c
#include <stdio.h>
#include <stdlib.h>
#define COLUMN 5
#define ROW 3

int find_min( int* arr, int len );
int find_max( int* arr, int len );

int main() {
    int matrix[ROW][COLUMN];
    // input
    int i, j;
    for ( i=0; i<ROW; i++ ) {
        for ( j=0; j<COLUMN; j++ ) {
            scanf("%d", &matrix[i][j]);
        }
    }
    // find min
    int min_num[COLUMN+ROW];
    int cnt = 0;
    for ( ; cnt<COLUMN+ROW; cnt++ ) {
        if ( cnt < 3 ) {    // find the min_num of every row
            int* tmp = (int*)malloc(5*sizeof(int));
            for ( i=0; i<COLUMN; i++ ) {
                tmp[i] = matrix[cnt][i];
            }
            min_num[cnt] = find_min(tmp, COLUMN);
            free(tmp);
        }
        else {  // find the min_num of every col
            int* tmp = (int*)malloc(3*sizeof(int));
            for ( i=0; i<ROW; i++ ) {
                tmp[i] = matrix[i][cnt-3];
            }
            min_num[cnt] = find_min(tmp, ROW);
            free(tmp);
        }
    }
    // find max
    printf("%d", find_max(min_num, COLUMN+ROW));
    return 0;
}

int find_min( int* arr, int len ) {
    int min = arr[0], i;
    for ( i=0; i<len; i++ ) {
        if (arr[i] < min) {
            min = arr[i];
        }
    }
    return min;
}

int find_max( int* arr, int len ) {
    int max = arr[0], i;
    for ( i=0; i<len; i++ ) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}
````

````c
#include <stdio.h>
#include <string.h>
#define CHAR_MAX 10

void DigiTran( int k, char* res );

int main() {
    int k;
    scanf("%d", &k);
    while ( k != -1 ) {
        char res[CHAR_MAX] = {'\0'};
        DigiTran(k, res);
        scanf("%d", &k);
    }
    return 0;
}

void DigiTran( int k, char* res ) {
    char str[CHAR_MAX];
    sprintf( str, "%d", k );
    int i = 0, cnt = 0, len = strlen(str);
    while ( i < len ) {
        if ( i+1 < len ) {
            if ( (str[i]-'0')*10+(str[i+1]-'0') < 26 ) {
                res[cnt] = 'A' + (str[i]-'0')*10+(str[i+1]-'0');
                cnt ++;
                i += 2;
            }
            else {
                res[cnt] = 'A' + (str[i]-'0');
                cnt++;
                i++;
            }
        }
        else {
            res[cnt] = 'A' + (str[i]-'0');
            i++;
        }
    }
    printf("%s ", res);
}
````

````c
#include <stdio.h>
#define NAME_LEN 20
#define FRUIT_NUM 6

typedef struct {
    int id;
    char name[NAME_LEN];
    double sv;
    double unit_p;
    double revenue;
} FruitInfo;

void set( FruitInfo* info );
void sort_fruit( FruitInfo* info );
void print( FruitInfo* info );


int main() {
    FruitInfo info[FRUIT_NUM];
    int i;
    for ( i=0; i<FRUIT_NUM; i++ ) {
        set(&info[i]);
    }
    sort_fruit(info);
    for ( i=0; i<FRUIT_NUM; i++ ) {
        print(&info[i]);
    }
    return 0;
}

void set( FruitInfo* info ) {
    scanf("%d %s %lf %lf", &(info->id), &(info->name), &(info->sv), &(info->unit_p));
    info->revenue = info->sv * info->unit_p;
}

void sort_fruit( FruitInfo* info ) {
    int i, j;
    for( i=0; i<FRUIT_NUM-1; i++ ) {
        for ( j=0; j<FRUIT_NUM-i-1; j++ ) {
            if ( info[j].revenue > info[j+1].revenue ) {
                FruitInfo tmp = info[j];
                info[j] = info[j+1];
                info[j+1] = tmp;
            }
        }
    }
}

void print( FruitInfo* info ) {
    printf("%d %s %.2f %.2f\n", info->id, info->name, info->sv, info->unit_p);
}
````
